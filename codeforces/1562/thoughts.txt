#include <bits/stdc++.h>
using namespace std;

#define int long long
#define forn(i, a, n) for (int i = a; i < n; i++)
#define MAXN 200100
#define MOD 1000000007

void solve() {
    int n;
    string s;
    cin >> n;
    cin >> s;
    int dp[n+1];
    forn(i, 0, n+1) dp[i] = 0;
    dp[n] = 1;
    for(int i = n-1; i>=1; i--) {
        int t = 0;
        for(int j = i + 1; j <= n; j++) {
            if(s[j-1] > s[i-1]) t = max(t, dp[j]);
        }
        dp[i] = n - i + 1 + t;
    }
    // forn(i, 1, n+1) cerr << dp[i] << " ";
    sort(dp, dp + n + 1);
    cout << dp[n] << endl;
}

int32_t main() {

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    // freopen("input.txt", "r", stdin);
    // freopen("output.txt", "w", stdout);
    
    int T = 1;
    cin >> T;
    while(T--) solve();

    return 0;
}
